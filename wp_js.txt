        // ---- InicializaciÃ³n basada en la URL (al recargar) ----
        document.addEventListener("DOMContentLoaded", () => {
            const hash = window.location.hash.substring(1);
            if (hash && document.getElementById('page-' + hash)) {
                window.history.replaceState({ page: hash }, '', '#' + hash);
                showPageUI(hash);
            } else {
                window.history.replaceState({ page: 'home' }, '', window.location.pathname);
                showHomeUI();
            }
        });

        // ---- NavegaciÃ³n SubpÃ¡ginas (Historial y Gestos Android) ----
        window.addEventListener('popstate', function (event) {
            if (event.state && event.state.page && event.state.page !== 'home') {
                showPageUI(event.state.page);
            } else {
                showHomeUI();
            }
        });

        function showHomeUI() {
            pages.forEach(p => p.classList.remove('active'));
            homePage.classList.add('active');
            window.scrollTo({ top: 0, behavior: 'instant' });
        }

        function showPageUI(pageId) {
            pages.forEach(p => p.classList.remove('active'));
            const target = document.getElementById('page-' + pageId);
            if (target) {
                target.classList.add('active');
                window.scrollTo({ top: 0, behavior: 'instant' });
            }
        }

        window.currentHimnoIndex = 0;
        function selectHimno(index, shouldReturnToPlayer = true, shouldScrollList = false) {
            const list = document.getElementById('xperia-himnos-list');
            if (!list) return;
            const entries = Array.from(list.querySelectorAll('.xperia-canto-entry'));
            const player = document.getElementById('xperia-himnos-player');

            if (!entries.length || !player) return;

            window.currentHimnoIndex = Math.max(0, Math.min(index, entries.length - 1));
            const activeEntry = entries[window.currentHimnoIndex];

            entries.forEach((entry, entryIndex) => {
                entry.classList.toggle('active', entryIndex === window.currentHimnoIndex);
            });

            if (player.getAttribute('src') !== activeEntry.dataset.embed) {
                player.src = activeEntry.dataset.embed;
            }

            if (shouldScrollList) {
                activeEntry.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
            }

            if (shouldReturnToPlayer) {
                scrollHimnosTop();
            }
        }

        function stepCantosSelection(direction) {
            selectCanto(currentCantoIndex + direction, false, true);

        function initCantosScroller() {
            const list = document.getElementById('xperia-cantos-list');

            list.dataset.ready = 'true';

        function selectCanto(index, shouldReturnToPlayer = true, shouldScrollList = false) {
            const list = document.getElementById('xperia-cantos-list');
            if (!list) return;
            const entries = Array.from(list.querySelectorAll('.xperia-canto-entry'));
            const player = document.getElementById('xperia-cantos-player');
            const openLink = document.getElementById('xperia-cantos-open-link');
            const title = document.getElementById('xperia-cantos-title');
            const artist = document.getElementById('xperia-cantos-artist');
            const counter = document.getElementById('xperia-cantos-counter');

            if (!entries.length || !player) return;

            currentCantoIndex = Math.max(0, Math.min(index, entries.length - 1));
            const activeEntry = entries[currentCantoIndex];

            entries.forEach((entry, entryIndex) => {
                entry.classList.toggle('active', entryIndex === currentCantoIndex);
            });

            if (player.getAttribute('src') !== activeEntry.dataset.embed) {
                player.src = activeEntry.dataset.embed;
            }
            if (openLink && openLink.getAttribute('href') !== activeEntry.dataset.link) {
                openLink.href = activeEntry.dataset.link;
            }
            if (title) title.textContent = activeEntry.dataset.title;
            if (artist) artist.textContent = activeEntry.dataset.artist;
            if (counter) counter.textContent = `${String(currentCantoIndex + 1).padStart(2, '0')} / ${String(entries.length).padStart(2, '0')}`;

            if (shouldScrollList) {
                activeEntry.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
            }

            if (shouldReturnToPlayer) {
                scrollMusicTop();
            }
        }

        function createSelectFunction(listId, playerId, titleId, artistId, counterId, cardId) {
            return function(index, shouldReturnToPlayer = true, shouldScrollList = false) {
                const list = document.getElementById(listId);
                if (!list) return 0;
                const entries = Array.from(list.querySelectorAll('.xperia-canto-entry'));
                const player = document.getElementById(playerId);
                const title = document.getElementById(titleId);
                const artist = document.getElementById(artistId);
                const counter = document.getElementById(counterId);

                if (!entries.length || !player) return 0;

                const currentIndex = Math.max(0, Math.min(index, entries.length - 1));
                const activeEntry = entries[currentIndex];

                entries.forEach((entry, entryIndex) => {
                    entry.classList.toggle('active', entryIndex === currentIndex);
                });

                if (player.getAttribute('src') !== activeEntry.dataset.embed) {
                    player.src = activeEntry.dataset.embed;
                }
                if (title) title.textContent = activeEntry.querySelector('.xperia-canto-title').textContent;
                if (artist) artist.textContent = activeEntry.querySelector('.xperia-canto-artist').textContent;
                if (counter) counter.textContent = `${String(currentIndex + 1).padStart(2, '0')} / ${String(entries.length).padStart(2, '0')}`;

                if (shouldScrollList) {
                    activeEntry.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
                }

                if (shouldReturnToPlayer) {
                    const card = document.getElementById(cardId);
                    if (card) {
                        const top = card.getBoundingClientRect().top + window.pageYOffset - 12;
                        window.scrollTo({ top, behavior: 'smooth' });
                    }
                }
                return currentIndex;
            };
        }

        // ---- LÃ³gica Reproductor Radio Custom ----
        let currentRadioAudio = null;
        let currentRadioIcon = null;

        function stopRadios() {
            if (currentRadioAudio) {
                currentRadioAudio.pause();
                if (currentRadioIcon) {
                    currentRadioIcon.classList.remove('fa-pause', 'fa-beat-fade', 'fa-spinner', 'fa-spin');
                    currentRadioIcon.classList.add('fa-play');
                }
                currentRadioAudio = null;
                currentRadioIcon = null;
            }
        }

        function toggleRadio(id, btn) {
            const audio = document.getElementById(id);
            const icon = btn.querySelector('i');

            if (currentRadioAudio && currentRadioAudio !== audio) {
                stopRadios();
            }

            if (audio.paused) {
                // Mostrar estado de carga
                icon.classList.remove('fa-play');
                icon.classList.add('fa-spinner', 'fa-spin');

                // Forzar reconexiÃ³n para streamings en vivo sin cachÃ©
                let baseSrc = audio.getAttribute('data-src');
                if (!baseSrc) {
                    baseSrc = audio.src;
                    audio.setAttribute('data-src', baseSrc);
                }
                // AÃ±adir un timestamp asegurando conexiÃ³n fresca
                audio.src = baseSrc.split('?')[0] + '?t=' + new Date().getTime();
                audio.load(); // Es importante hacer load despues de setear el src

                let playPromise = audio.play();
                if (playPromise !== undefined) {
                    playPromise.then(() => {
                        icon.classList.remove('fa-spinner', 'fa-spin');
                        icon.classList.add('fa-pause', 'fa-beat-fade');
                        currentRadioAudio = audio;
                        currentRadioIcon = icon;
                    }).catch(e => {
                        console.error("Error reproduciendo el audio:", e);
                        icon.classList.remove('fa-spinner', 'fa-spin');
                        icon.classList.add('fa-play');
                        // Intento de fallback
                        setTimeout(() => audio.play(), 500);
                    });
                }
            } else {
                stopRadios();
            }
        }

        // Refuerzo: detener radios si el usuario pone el cursor sobre un reproductor de video
        document.addEventListener('DOMContentLoaded', () => {
            initCantosScroller();
            initNewScrollers();
            document.querySelectorAll('iframe').forEach(ifr => {
                ifr.addEventListener('mouseenter', stopRadios);
                ifr.addEventListener('touchstart', stopRadios);
            });

            // ==== CONEXION CON GITHUB PARA VIDEO EN VIVO ====
            const configUrl = 'https://raw.githubusercontent.com/controlvilleda-star/youtube-bridge/refs/heads/master/video.json?nocache=' + new Date().getTime();
            fetch(configUrl).then(response => {
                if (!response.ok) throw new Error('Error al leer JSON');
                return response.json();
            }).then(data => {
                if (data && data.videoId) {
                    const iframe = document.getElementById('iframe-online');
                    if (iframe) {
                        let conector = data.videoId.includes('?') ? '&' : '?';
                        let newUrl = 'https://www.youtube.com/embed/' + data.videoId + conector + 'autoplay=1&mute=0';
                        iframe.setAttribute('data-src', newUrl);
                        
                        // Si el panel de video ya está abierto, cambiamos el video en vivo al instante
                        const panelOnline = document.getElementById('panel-online');
                        if (panelOnline && !panelOnline.classList.contains('hidden')) {
                            if (iframe.src && !iframe.src.includes(data.videoId)) {
                                iframe.src = newUrl;
                            }
                        }
                    }
                }
            }).catch(e => console.error('Error video:', e));
        });
